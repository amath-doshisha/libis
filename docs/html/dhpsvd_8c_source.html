48  C=dvec_dot(m,z,u); // C=z'*u

58  dvec_sub(m,u,&Hu); // u=u-H(1:m);
59  dvec_sub(n,v,&Hv); // v=v-H((m+1):end);
60  }

63  }

66  dvec_normalize_sgn(n,v); // v=v/sqrt(v'*v)

71  e=dvec_norm_max(m+n,H); // e=max(abs(H))
72  if(isnan(u[0])) { ret=DHPSVD_NAN; e=1; }
73  if(debug>0){ printf("[dhpsvd_1pair] done=%d ret=%d step=%02d log2(e)=%+6.2f sigma=%+.16e\n",done,ret,step,log(fabs(e))/log(2),sigma); }
74  // free

84 }
