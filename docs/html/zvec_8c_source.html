810  // open file
811  if((fid=fopen(fname,"r"))==0){ fclose(fid); zx=NULL; (*n)=0; }
812  else{
813  // read header
814  l=strlen("zvec");

818  // read size
819  k=fread(n,sizeof(int),1,fid);

821  // allocate
822  zx=zvec_allocate((*n));
823  // read data
824  k=fread(zx,sizeof(dcomplex),(*n),fid);
825  if(k!=(size_t)(*n)){ ERROR_AT; printf("Failed to load the data from the file '%s'.\n",fname); exit(0); }
826  // close
827  fclose(fid);
828  }else if(k==(size_t)l && strncmp(buf,"dvec",l)==0){ /* dvec */
829  fclose(fid); // close
830  dx=dvec_bin_load(n,fname); // load
831  zx=zvec_allocate((*n)); // allocate
