94 void csolve_lu_decomp(int n, cmulti **A, int LDA, int *p, int *info)
95 {
96  int prec,ret=0,i,j,k,l;
97  cmulti *a=NULL;
98  rmulti *c=NULL,*value=NULL;
99  // allocate
100  prec=cmat_get_prec_max(n,n,A,LDA);
101  a=callocate_prec(prec);
102  c=rallocate_prec(prec);
103  value=rallocate_prec(prec);
104  // LU分解
105  for(k=0; k<n; k++){
106  // pivot select
107  cabsv(value,MA(k,k));
108  for(l=k,j=k+1; j<n; j++){
109  cabsv(c,MA(j,k));
110  if(rgt(c,value)) { l=j; rcopy(value,c); } // value=c
111  }
112  // エラー処理
113  if(ris_zero(value)) { ret=l+1; break; }
114  p[k]=l;
115  // swap A(k,k:end)<-> A(l,k:end)
116  if(l!=k){
117  for(j=k; j<n; j++){
118  cswap(MA(k,j),MA(l,j));
119  }
120  }
121  // L行列とU行列の作成
122  for(i=k+1; i<n; i++){
123  cdiv(a,MA(i,k),MA(k,k));
124  // L行列 A(i,k)=a
125  ccopy(MA(i,k),a);
126  // U行列 A(i,k+1:end)+=(-a)*A(k,k+1:end)
127  for(j=k+1; j<n; j++){
128  csub_mul(MA(i,j),a,MA(k,j));
129  }
130  }
131  }
132  // done
