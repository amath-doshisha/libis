585 void irvec_func(rmulti *y0, rmulti *y1, func_t *f, int n, rmulti **x0, rmulti **x1)

589  RA2(a0,y0,y1); RA2(a1,y0,y1); RA2(b0,y0,y1); RA2(b1,y0,y1); RA2(z0,y0,y1); RA2(z1,y0,y1);
590  rset_nan(z0); rset_nan(z1);
591  if(f==NULL) { FUNC_ERROR_ARG1("irvec_func",f); }
592  else if(func_is(f,"inf")) { rset_inf(z0,1); rset_inf(z1,1); }
593  else if(func_is_zero(f)) { irset_d(z0,z1,0); }
594  else if(func_is_one(f)) { irset_d(z0,z1,1); }
595  else if(func_is_bigint(f)) { irset_bigint(z0,z1,func_bigint_p(f)); }
596  else if(func_is_real(f)) { ircopy(z0,z1,func_real_p(f),func_real_p(f)); }
597  else if(func_is_var(f)) {
598  irset_d(z0,z1,1);
599  for(i=0; i<func_var_size(f); i++){
600  if(func_var_pow(f,i)!=0){
601  if(0<=func_var_num(f,i) && func_var_num(f,i)<n){
602  irpow_si(a0,a1,x0[func_var_num(f,i)],x1[func_var_num(f,i)],func_var_pow(f,i));
603  }else{ rset_nan(a0); rset_nan(a1); }
604  irmul(z0,z1,z0,z1,a0,a1);
605  }
606  }
607  }
608  else if(func_is_add(f)) {
609  irset_d(z0,z1,0);
610  for(i=0; i<func_asize(f); i++){
611  irvec_func(a0,a1,func_aget(f,i),n,x0,x1);
612  iradd(z0,z1,z0,z1,a0,a1);
613  }
614  }
615  else if(func_is_mul(f)) {
616  irset_d(z0,z1,1);
617  for(i=0; i<func_asize(f); i++){
