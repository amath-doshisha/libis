526  rset_nan(z0); rset_nan(z1);
527  if(f==NULL) { FUNC_ERROR_ARG1("irvec_func",f); }
528  else if(func_is(f,"inf")) { rset_inf(z0,1); rset_inf(z1,1); }
529  else if(func_is_zero(f)) { irset_d(z0,z1,0); }
530  else if(func_is_one(f)) { irset_d(z0,z1,1); }
531  else if(func_is_bigint(f)) { irset_bigint(z0,z1,func_bigint_p(f)); }
532  else if(func_is_real(f)) { ircopy(z0,z1,func_real_p(f),func_real_p(f)); }
533  else if(func_is_var(f)) {
534  irset_d(z0,z1,1);
535  for(i=0; i<func_var_size(f); i++){
536  if(func_var_pow(f,i)!=0){
537  if(0<=func_var_num(f,i) && func_var_num(f,i)<n){
538  irpow_si(a0,a1,x0[func_var_num(f,i)],x1[func_var_num(f,i)],func_var_pow(f,i));
539  }else{ rset_nan(a0); rset_nan(a1); }
540  irmul(z0,z1,z0,z1,a0,a1);
541  }
542  }
543  }

545  irset_d(z0,z1,0);
546  for(i=0; i<func_asize(f); i++){

549  }

556  }
