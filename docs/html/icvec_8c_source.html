495 int icvec_func(cmulti *y0, cmulti *y1, func_t *f, int n, cmulti **x0, cmulti **x1)
496 {
497  int prec,i,e=0;
498  cmulti *a0=NULL,*a1=NULL,*b0=NULL,*b1=NULL,*z0=NULL,*z1=NULL;
499  prec=MAX2(cget_prec(y0),cget_prec(y1));
500  CA(a0,prec); CA(a1,prec); CA(b0,prec); CA(b1,prec); CA(z0,prec); CA(z1,prec);
501  cset_nan(z0); cset_nan(z1);
502  if(f==NULL) { FUNC_ERROR_ARG1("icvec_func",f); }
503  else if(func_is(f,"nan")) { cset_nan(z0); cset_nan(z1); }
504  else if(func_is(f,"inf")) { cset_inf(z0,1,1); cset_inf(z1,1,1); }
505  else if(func_is_zero(f)) { e+=icset_dd(z0,z1,0,0); }
506  else if(func_is_one(f)) { e+=icset_dd(z0,z1,1,0); }
507  else if(func_is_bigint(f)) { e+=icset_bigint(z0,z1,func_bigint_p(f)); }
508  else if(func_is_real(f)) { e+=ccopy_r(z0,func_real_p(f)); e+=ccopy_r(z1,func_real_p(f)); }
509  else if(func_is_complex(f)){ e+=ccopy(z0,func_complex_p(f)); e+=ccopy(z1,func_complex_p(f)); }
510  else if(func_is_var(f)) {
511  icset_dd(z0,z1,1,0);
512  for(i=0; i<func_var_size(f); i++){
513  if(func_var_pow(f,i)!=0){
514  if(0<=func_var_num(f,i) && func_var_num(f,i)<n){
515  e+=icpow_si(a0,a1,x0[func_var_num(f,i)],x1[func_var_num(f,i)],func_var_pow(f,i));
516  }else{ cset_nan(a0); cset_nan(a1); }
517  e+=icmul(z0,z1,z0,z1,a0,a1);
518  }
519  }
520  }
521  else if(func_is_add(f)) {
522  icset_dd(z0,z1,0,0);
523  for(i=0; i<func_asize(f); i++){
524  e+=icvec_func(a0,a1,func_aget(f,i),n,x0,x1);
525  e+=icadd(z0,z1,z0,z1,a0,a1);
526  }
527  }
528  else if(func_is_mul(f)) {
529  icset_dd(z0,z1,1,0);
530  for(i=0; i<func_asize(f); i++){
531  e+=icvec_func(a0,a1,func_aget(f,i),n,x0,x1);
532  e+=icmul(z0,z1,z0,z1,a0,a1);
533  }
534  }
