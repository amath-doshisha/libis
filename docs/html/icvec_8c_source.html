506 void icvec_func(cmulti *y0, cmulti *y1, func_t *f, int n, cmulti **x0, cmulti **x1)

510  prec=MAX2(cget_prec(y0),cget_prec(y1));
511  CA(a0,prec); CA(a1,prec); CA(b0,prec); CA(b1,prec); CA(z0,prec); CA(z1,prec);
512  cset_nan(z0); cset_nan(z1);
513  if(f==NULL) { FUNC_ERROR_ARG1("icvec_func",f); }
514  else if(func_is(f,"nan")) { cset_nan(z0); cset_nan(z1); }
515  else if(func_is(f,"inf")) { cset_inf(z0,1,1); cset_inf(z1,1,1); }
516  else if(func_is_zero(f)) { icset_dd(z0,z1,0,0); }
517  else if(func_is_one(f)) { icset_dd(z0,z1,1,0); }
518  else if(func_is_bigint(f)) { icset_bigint(z0,z1,func_bigint_p(f)); }
519  else if(func_is_real(f)) { ccopy_r(z0,func_real_p(f)); ccopy_r(z1,func_real_p(f)); }
520  else if(func_is_complex(f)){ ccopy(z0,func_complex_p(f)); ccopy(z1,func_complex_p(f)); }
521  else if(func_is_var(f)) {
522  icset_dd(z0,z1,1,0);
523  for(i=0; i<func_var_size(f); i++){
524  if(func_var_pow(f,i)!=0){
525  if(0<=func_var_num(f,i) && func_var_num(f,i)<n){
526  icpow_si(a0,a1,x0[func_var_num(f,i)],x1[func_var_num(f,i)],func_var_pow(f,i));
527  }else{ cset_nan(a0); cset_nan(a1); }
528  icmul(z0,z1,z0,z1,a0,a1);
529  }
530  }
531  }
532  else if(func_is_add(f)) {
533  icset_dd(z0,z1,0,0);
534  for(i=0; i<func_asize(f); i++){
535  icvec_func(a0,a1,func_aget(f,i),n,x0,x1);
536  icadd(z0,z1,z0,z1,a0,a1);
537  }
538  }
539  else if(func_is_mul(f)) {
