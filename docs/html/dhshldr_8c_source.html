12 void dhouseholder_vec(int n, int k, double *h, double *alpha, const double *x)

14  // function [h,alpha]=householder_vec(x,k)
15  // y=H*x, H=I-2*h*h'/(h'*h), h=x-y
16  // i<k y(i)=x(i) h(i)=0
17  // i=k y(k)=s*eta h(k)=x(k)-y(k)=x(k)-s*eta=-s*xi
18  // i>k y(i)=0 h(i)=x(i)
19  // eta=norm2(x(k:end))
20  // s=sgn(x(k))
21  // xi=-h(k)/s=-(x(k)-s*eta)/s=eta-x(k)/s=eta-|x(k)|
22  // =(eta^2-|x(k)|^2)/(eta+|x(k|)=(|x(k+1)|^2+..+|x(n)|^2)/(eta+|x(k)|)
23  // h'*h=2*xi*eta
24  // alpha=2/(h'*h)=1/(xi*eta)
25  double xi,eta,s=1;
26  //----------- norm
27  xi=dvec_dot(n-k-1,&x[k+1],&x[k+1]); // xi=sum(abs(x((k+1):end)).^2);
28  eta=sqrt(x[k]*x[k]+xi);
29  if(eta==0) xi=eta-fabs(x[k]);
30  else xi=xi/(fabs(x[k])+eta);
31  if(x[k]<0) s=-1;
32  //----------- h
33  dvec_zeros(k,h);
34  dvec_copy(n-k-1,&h[k+1],&x[k+1]); // h((k+1):end)=x((k+1):end);
35  h[k]=-s*xi;
36  //----------- alpha
37  if(xi==0 || eta==0) (*alpha)=0;
38  else (*alpha)=1.0/(xi*eta);
39 }
