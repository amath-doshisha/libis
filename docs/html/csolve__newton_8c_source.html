544 int csolve_newton(int m, cmulti **x, func_t *fF, int step_max, int debug)
545 {

547  int info,n,ret;
548  func_t *fJ=NULL;

550  rmulti *eta=NULL,*eta_pre=NULL,*mu=NULL,*b=NULL,*a=NULL,*emax=NULL;
551  // init
552  CVA(F,m); CMA(J,m,m); RA(eta); RA(eta_pre); RA(mu); RA(b); RA(a);
553  // begin
554  if(step_max<0){ step_max=100; }
555  fJ=func_grad(func_retain(fF),func_var1_list(m));
556  if(debug>0){
557  mpfr_printf("[%s] step_max=%d\n",name,step_max);
558  printf("[%s] F(x)=",name); func_print(fF); printf("\n");
559  printf("[%s] F'(x)=",name); func_print(fJ); printf("\n");
560  }
561  // compute
562  if(cvec_has_nan(m,x)){ cvec_set_rand(m,x,2,-1); }
563  n=0; rset_d(eta_pre,0);
564  if(debug>0){ mpfr_printf("[%s] begin x[%d]\n",name,n,n); }
565  do{
566  // a=|x|
567  cvec_max_abs(a,m,x);
568  // F=F(x)
569  cvec_func_list(m,F,fF,m,x);
570  // b=|F|

574  // if F=0, then...
575  if(eq_rd(b,0)){ status='='; }

578  cmat_func_list2(m,m,J,m,fJ,m,x);

580  csolve(m,1,F,m,J,m,&info);
581  if(info){

583  status='J';
584  rset_nan(eta); rset_nan(mu);
585  }else{
586  // eta=|J\F|
587  cvec_max_abs(eta,m,F);
588  // mu=eta/eta_pre
589  rdiv_rr(mu,eta,eta_pre);
590  // set status
591  if (n>=step_max) { status='/'; }
592  else if(status==' ' && le_rd(mu,1e-2)){ status='o'; }
593  else if(status=='o' && ge_rd(mu,1)) { status='x'; }
594  }
595  }

598  info=csolve_krawczyk(m,e,x,fF,0);
599  cvec_max_abs(emax,m,e);

601  else { mpfr_printf("[%s] n=%d [%c] e=%.0Re |x[%d]|=%.1Re |F(x[%d])|=%.1Re |x[%d]-x[%d]|=%.1Re mu=%.1Re\n",name,n,status,emax,n,a,n,b,n+1,n,eta,mu); }

603  }
604  // update x

607  cvec_sub_cvec(m,x,x,F);
608  // next
609  n++;
610  rswap(eta,eta_pre);
611  }
612  }while(status==' ' || status=='o');
613  // value
614  if (status=='/' || status=='=' || status=='x'){ ret=0; }
615  else if(status=='o' || status==' ' || status=='J'){ ret=1; }
616  else { ret=1; }

618  // done
619  CVF(F,m); CMF(J,m,m); RF(eta); RF(eta_pre); RF(mu); RF(b); RF(a);
620  fJ=func_del(fJ);
621  return ret;
622 }

676  cvec_lec(m,X0,H0) && cvec_lec(m,H1,X1)){ ret=0; }
677  else { ret=1; }
678  // done
679  CMF(L,m,m); CMF(L0,m,m); CMF(L1,m,m); CMF(R,m,m); CMF(M0,m,m); CMF(M1,m,m);

681  CVF(T0,m); CVF(T1,m); CVF(r0,m); CVF(r1,m); CVF(H0,m); CVF(H1,m);
