42  radd_rr(eta,axk,xi); // eta=|x[k]|^2+...
43  rsqrt_r(axk,axk); // axk=|x[k]|
44  rsqrt_r(eta,eta); // eta=sqrt(|x[k]|^2+...)

46  else{ // xi=xi/(|x(k)|+eta)
47  radd_rr(zeta,axk,eta);
48  rdiv_rr(xi,xi,zeta);
49  }
50  //----------- h

52  cvec_copy_cvec(n-k-1,&h[k+1],&x[k+1]); // h((k+1):end)=x((k+1):end);

55  }else{

57  cmul_cr(h[k],x[k],zeta); // h[k]=zeta*x[k];
58  }
59  //----------- alpha

61  rset_d(alpha,0);

64  rinv_r(alpha,alpha);
65  }
